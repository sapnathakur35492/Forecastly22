x += xC(`CAGR (${years[0]}-${years[years.length - 1]})`, 'hdrC'); x += `</Row>`; x += 
`<Row>${xCi('Revenue (' + curr + ')', 'label')}`; years.forEach(() => { x += xC(XX, 'data'); }); x += xC(XXP, 
'data'); x += `</Row>`; x += `<Row>${xMi(SRC, 'src', nc - 1)}</Row><Row></Row><Row></Row>`; return 
x; } 
    function xC(v, s) { return `<Cell ss:StyleID="${s}"><Data ss:Type="String">${v}</Data></Cell>`; }

ss:Type="String">${v}</Data></Cell>`; } 
    function xMi(v, s, ma) { return `<Cell ss:Index="2" ss:StyleID="${s}" ss:MergeAcross="${ma}"><Data

    function renderPricing() { 
        const container = document.getElementById('pricingCards'); 
        container.innerHTML = ''; 
        Object.entries(pricingData).forEach(([key, plan]) => { 
            const card = document.createElement('div'); 
            card.className = `price-option ${currentPackage === key ? 'active' : ''}`; 
            let tags = plan.tags.map(t => `<span class="plan-feat-tag">${t}</span>`).join('') + 
plan.disabled.map(t => `<span class="plan-feat-tag disabled">${t}</span>`).join(''); 
            card.innerHTML = ` 
                ${plan.popular ? '<div class="popular-badge">Most Popular</div>' : ''} 
                <div class="price-header"><span class="plan-name">${plan.emoji} ${plan.name}</span><span 
class="price">${plan.price}</span></div> 
                <div class="price-desc">${plan.desc}</div> 
                <div class="plan-features">${tags}</div> 
                <div style="display: flex; gap: 0.5rem; margin-top: 0.8rem;"> 
                    <button class="demo-btn" data-package="${key}" style="flex:1; background: var(--gray-100); 
border: 1px solid var(--gray-200); padding: 0.5rem; border-radius: 40px; font-size: 0.7rem; font-weight: 
600; cursor: pointer;">📥 Demo Excel</button> 
                    <button class="buy-btn" data-package="${key}" style="flex:1; background: var(--accent); 
color: white; border: none; padding: 0.5rem; border-radius: 40px; font-size: 0.7rem; font-weight: 600; 
cursor: pointer;">💰 Buy Now</button> 
                </div> 
            `; 
            card.querySelector('.demo-btn').addEventListener('click', (e) => { 
                e.stopPropagation(); 
                currentPackage = key; 
                document.querySelectorAll('.price-option').forEach(c => c.classList.remove('active')); 
                card.classList.add('active'); 
                applyR(); 
                buildExcel(); 
            }); 
            card.querySelector('.buy-btn').addEventListener('click', (e) => { 
                e.stopPropagation(); 
                const title = document.getElementById('reportTitleInput').value.trim(); 
                if (!title) { alert("Please enter a market report title before purchasing."); return; } 
                currentReportTitleForPayment = title; 
                currentPackageForPayment = key; 
                generatedOrderId = 'MP-' + Date.now() + '-' + Math.floor(Math.random() * 10000); 
                showPage('dashboard'); 
                renderDashboardContent('orders'); 
            }); 
            card.addEventListener('click', () => { 
                currentPackage = key; 
                document.querySelectorAll('.price-option').forEach(c => c.classList.remove('active')); 
                card.classList.add('active'); 
                applyR(); 
            }); 
            container.appendChild(card); 
        }); 
    }

    function initBuilder() { 
        if (builderInited) return; 
        builderInited = true; 
        renderRegions(); 
        renderPricing(); 
        updateChips(); 
        applyR(); 
        updateForecastOptions(); 
        const q = document.getElementById('reportTitleInput').value.trim(); 
        if (q) refreshSeg(); 
        document.getElementById('reportTitleInput').addEventListener('input', refreshSeg); 
        document.getElementById('baseYear').addEventListener('change', updateForecastOptions); 
        document.getElementById('addCountryBtn').addEventListener('click', () => { 
            if (currentPackage === 'basic') { alert("Upgrade to Pro/Enterprise."); return; } 
            const inp = document.getElementById('customCountryInput'), regSel = 
document.getElementById('customRegionSelect'); 
            let v = inp.value.trim(); 
            if (!v) return; 
            v = v.charAt(0).toUpperCase() + v.slice(1); 
            if (customCountries.hasOwnProperty(v) || selectedCountries.has(v)) { alert("Already selected."); 
return; } 
            customCountries[v] = regSel.value; 
            updateChips(); 
            inp.value = ''; 
        }); 
        document.getElementById('generateExcelBtn').addEventListener('click', buildExcel); 
        setTimeout(() => { 
            ["USA", "Germany", "China", "India", "Brazil", "UAE", "Rest of Europe", "Rest of APAC", "Rest of 
LATAM", "Rest of MEA"].forEach(c => { 
                if (countryCheckboxes[c]) { selectedCountries.add(c); countryCheckboxes[c].checked = true; } 
            }); 
            updateChips(); 
        }, 60); 
    }
